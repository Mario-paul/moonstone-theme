 .panel-top, .panel-bottom, .panel-left, .panel-right {

    }

.menu {
  background-color: rgba(46, 46, 46, 0.59); } /* menu bg color & alpha */

.menu-favorites-box {
  background-color: rgba(64, 64, 64, 0.75); /* menu favorites box bg color & alpha */
 }
/* menu search box */

    }

.switcher-list {

    }

    }

/* Grouped window list thumbnail */
